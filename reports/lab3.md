```rust
impl PartialOrd for Stride {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        let res = self.0 < other.0;
        if |self.0 - other.0| > BIG_STRIDE/2 {
            -res
        } else {
            res
        }
    }
}
```
